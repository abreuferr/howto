TOPOLOGIA
# haproxy - 192.168.0.15
# web1 - 192.168.0.16
# web2 - 192.168.0.17

INSTALACAO
# Instalacao do HAProxy
#
$ sudo aptitude install haproxy -y -d

CONFIGURACAO do HAPROXY
# Ativacao do HAProxy
#
$ sudo vi /etc/default/haproxy
		ENABLED=1

# Backup de seguranca
#
$ sudo mv /etc/haproxy/haproxy.cfg /etc/haproxy/haproxy.cfg.BKP

# Configuracao do HAProxy
#
$ sudo vi /etc/haproxy/haproxy.cfg
	global
	log /dev/log   local0
	log 127.0.0.1   local1 notice
	maxconn 4096
	user haproxy
	group haproxy
	daemon

	defaults
	log     global
	mode    http
	option  httplog
	option  dontlognull
	retries 3
	option redispatch
	maxconn 2000
	timeout connect    5000
	timeout client     50000
	timeout server     50000

	listen webfarm 0.0.0.0:80
	mode http
	stats enable
	stats uri /haproxy?stats
	balance roundrobin
	option httpclose
	option forwardfor
	server web1 192.168.0.16:80 check
	server web2 192.168.0.17:80 check

# Restart
$ sudo /etc/init.d/haproxy restart
