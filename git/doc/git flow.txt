# Exemplo 01 (Default)
#

# branch "main"
git checkout main

# crio e acesso a branch "develop"
git checkout -b develop

# crio e acesso a branch "feature_branch"
git checkout -b feature_branch

# work happens on feature branch

# volto para a branch "develop"
git checkout develop

# merge da branch "feature_branch" na branch "develop"
git merge feature_branch

# volto para a branch "main"
git checkout main

# merge da branch "develop" na branch "main"
git merge develop

# apago a branch "feature_branch"
git branch -d feature_branch

#
# Exemplo 02 (Hotfix)
#

git checkout main
git checkout -b hotfix_branch

# work is done commits are added to the hotfix_branch

git checkout main
git merge hotfix_branch
git checkout develop
git merge hotfix_branch
git branch -D hotfix_branch
